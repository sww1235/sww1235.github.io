wget gz xrandr make thefuck clang libX11-devel xtools libxft libXft-devel libXinerama-devel xorg-minimal xorg-fonts xterm

qemu install

sudo ln -s /etc/sv/libvirtd/ /var/service sudo ln -s /etc/sv/virtlogd/ /var/service/ sudo ln -s /etc/sv/virtlockd/ /var/service/ sudo usermod -a -G libvirt toxic

sudo sv start libvirtd
